Task 3 file:-
change sample.yaml file and now change name according to name given in your lab

  # sandbox-metadata-test.yaml
  apiVersion: apps/v1
  kind: Deployment
  metadata:
    name: unrestricted
    labels:
      app: unrestricted
  spec:
    replicas: 1
    selector:
      matchLabels:
        app: unrestricted
    template:
      metadata:
        labels:
          app: unrestricted
      spec:
        affinity:
          nodeAffinity:
            requiredDuringSchedulingIgnoredDuringExecution:
              nodeSelectorTerms:
              - matchExpressions:
                - key: sandbox.gke.io/runtime
                  operator: In
                  values:
                  - gvisor
        tolerations:
          - effect: NoSchedule
            key: sandbox.gke.io/runtime
            operator: Equal
            value: gvisor
        containers:
        - name: unrestricted
          image: busybox
          command: ["/bin/sleep","10000"]
Run command :-

kubectl apply -f sample.yaml
